<script>
const WEBHOOK_URL = https://saad92005.app.n8n.cloud/webhook/attendance-punch;
const DB_NAME = 'AttendTrackDB';
const STORE = 'punches';
let db, currentLat = null, currentLng = null, currentAcc = null;

function openDB(){
  return new Promise((res,rej)=>{
    const req = indexedDB.open(DB_NAME,1);
    req.onupgradeneeded = e => {
      const d = e.target.result;
      if(!d.objectStoreNames.contains(STORE)){
        const s = d.createObjectStore(STORE,{keyPath:'id',autoIncrement:true});
        s.createIndex('synced','synced',{unique:false});
      }
    };
    req.onsuccess = e => res(e.target.result);
    req.onerror = ()=>rej('DB error');
  });
}

function savePunch(punch){
  return new Promise((res,rej)=>{
    const tx = db.transaction(STORE,'readwrite');
    const req = tx.objectStore(STORE).add(punch);
    req.onsuccess = ()=>res(req.result);
    req.onerror = ()=>rej('Save error');
  });
}

function getPendingPunches(){
  return new Promise((res)=>{
    const tx = db.transaction(STORE,'readonly');
    const idx = tx.objectStore(STORE).index('synced');
    const req = idx.getAll(0);
    req.onsuccess = ()=>res(req.result);
    req.onerror = ()=>res([]);
  });
}

function markSynced(id){
  return new Promise((res)=>{
    const tx = db.transaction(STORE,'readwrite');
    const store = tx.objectStore(STORE);
    const req = store.get(id);
    req.onsuccess = ()=>{
      const rec = req.result;
      rec.synced = 1;
      store.put(rec);
      res();
    };
  });
}

function getTodayPunches(){
  return new Promise((res)=>{
    const today = new Date().toISOString().slice(0,10);
    const tx = db.transaction(STORE,'readonly');
    const req = tx.objectStore(STORE).getAll();
    req.onsuccess = ()=>res((req.result||[]).filter(p=>p.timestamp.startsWith(today)));
    req.onerror = ()=>res([]);
  });
}

function startGPS(){
  if(!navigator.geolocation){document.getElementById('gps-status').textContent='Not supported';return;}
  navigator.geolocation.watchPosition(pos=>{
    currentLat = pos.coords.latitude.toFixed(6);
    currentLng = pos.coords.longitude.toFixed(6);
    currentAcc = Math.round(pos.coords.accuracy);
    document.getElementById('gps-lat').textContent = currentLat;
    document.getElementById('gps-lng').textContent = currentLng;
    document.getElementById('gps-acc').textContent = currentAcc+'m';
    document.getElementById('gps-status').textContent = currentAcc < 50 ? 'Good' : 'Acquiring...';
    reverseGeocode(currentLat,currentLng);
  },{enableHighAccuracy:true,timeout:15000,maximumAge:30000});
}

function reverseGeocode(lat,lng){
  fetch(`https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lng}&format=json`)
    .then(r=>r.json())
    .then(d=>{
      const a = d.address;
      const label = [a.road||a.suburb||'',a.city||a.town||a.village||''].filter(Boolean).join(', ');
      document.getElementById('gps-addr').textContent = label||'Location found';
    }).catch(()=>{
      document.getElementById('gps-addr').textContent = 'Address unavailable (offline)';
    });
}

async function punch(type){
  const emp = document.getElementById('staff-select').value;
  if(!emp){alert('Please select your name first.');return;}
  if(!currentLat){alert('GPS not ready yet. Please wait a moment.');return;}
  const deviceId = getDeviceId();
  const record = {
    employeeId: emp,
    type: type,
    timestamp: new Date().toISOString(),
    latitude: parseFloat(currentLat),
    longitude: parseFloat(currentLng),
    accuracy: currentAcc,
    deviceId: deviceId,
    synced: 0,
    localId: Date.now()
  };
  await savePunch(record);
  updateBadge();
  renderToday();
  if(navigator.onLine) syncNow();
}

async function syncNow(){
  if(!navigator.onLine){alert('No internet. Punches saved locally and will sync automatically.');return;}
  const pending = await getPendingPunches();
  if(pending.length===0){updateBadge();return;}
  let synced=0;
  for(const p of pending){
    try{
      const res = await fetch(WEBHOOK_URL,{
        method:'POST',
        headers:{'Content-Type':'application/json'},
        body:JSON.stringify(p)
      });
      if(res.ok){await markSynced(p.id);synced++;}
    }catch(e){console.log('Sync failed for',p.id,e);}
  }
  alert(synced>0?`Synced ${synced} punch(es) successfully!`:'Sync failed — will retry automatically.');
  updateBadge();
  renderToday();
}

async function updateBadge(){
  const pending = await getPendingPunches();
  const badge = document.getElementById('sync-badge');
  if(pending.length>0){
    badge.textContent=`${pending.length} pending`;
    badge.className='sync-badge pending';
  }else{
    badge.textContent='Synced';
    badge.className='sync-badge';
  }
}

async function renderToday(){
  const punches = await getTodayPunches();
  const list = document.getElementById('punch-list');
  if(punches.length===0){list.innerHTML='<p style="color:#999;font-size:13px;text-align:center;padding:20px">No punches today</p>';return;}
  list.innerHTML=punches.reverse().map(p=>`
    <div class="punch-row">
      <div>
        <span class="status-dot ${p.synced?'dot-green':'dot-amber'}"></span>
        <span class="punch-type ${p.type.toLowerCase()}">${p.type}</span>
      </div>
      <div style="text-align:right;color:#666">
        <div>${new Date(p.timestamp).toLocaleTimeString('en-PK',{hour:'2-digit',minute:'2-digit'})}</div>
        <div style="font-size:11px">${p.latitude},${p.longitude}</div>
      </div>
    </div>
  `).join('');
}

function getDeviceId(){
  let id=localStorage.getItem('attendtrack_device_id');
  if(!id){id='DEV_'+Math.random().toString(36).substr(2,9).toUpperCase();localStorage.setItem('attendtrack_device_id',id);}
  return id;
}

window.addEventListener('online',()=>{document.getElementById('offline-bar').classList.remove('show');syncNow();});
window.addEventListener('offline',()=>document.getElementById('offline-bar').classList.add('show'));

openDB().then(d=>{db=d;startGPS();updateBadge();renderToday();});
if('serviceWorker' in navigator)navigator.serviceWorker.register('sw.js');
if(!navigator.onLine)document.getElementById('offline-bar').classList.add('show');
